 * Suporta càrrega condicional, actualitzacions parcials i control de versions.
 */

const STORAGE_KEY = 'LAST_UPDATED_TIMESTAMP';

function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheets()[0];
  
  // Acció 1: Només comprovar si hi ha canvis (molt ràpid)
  if (e && e.parameter && e.parameter.action === 'check') {
    const lastUpdate = PropertiesService.getScriptProperties().getProperty(STORAGE_KEY) || "0";
    return ContentService.createTextOutput(JSON.stringify({timestamp: lastUpdate}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // Acció 2: Càrrega completa de dades
  const data = sheet.getDataRange().getValues();
  if (data.length === 0 || (data.length === 1 && data[0][0] === "")) {
    return ContentService.createTextOutput("[]").setMimeType(ContentService.MimeType.JSON);
  }
  
  const results = data.map(row => ({ d: row[0], m: row[1], c: row[2], k: row[3] }));
  return ContentService.createTextOutput(JSON.stringify(results))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const payload = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheet = ss.getSheets()[0];
    
    if (payload.type === 'partial') {
      // Actualització parcial d'una sola fila (índex basat en 1)
      const rowIndex = payload.index + 1;
      const entry = payload.data;
      sheet.getRange(rowIndex, 1, 1, 4).setValues([[entry.d, entry.m, entry.c, entry.k]]);
    } else {
      // Actualització completa (per defecte o si passem array directament)
      const json = Array.isArray(payload) ? payload : payload.data;
      sheet.clear();
      if (json && json.length > 0) {
        const rows = json.map(r => [r.d, r.m, r.c, r.k]);
        sheet.getRange(1, 1, rows.length, 4).setValues(rows);
      }
    }
    
    // Actualitzem la marca de temps de l'últim canvi
    const now = new Date().getTime().toString();
    PropertiesService.getScriptProperties().setProperty(STORAGE_KEY, now);
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", timestamp: now}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
